function filterVideos() { const search = document.getElementById("search").value.toLowerCase(); const category = document.getElementById("categoryFilter").value; const language = document.getElementById("languageFilter").value; const cards = document.querySelectorAll(".video-card"); cards.forEach(card => { const title = card.dataset.title.toLowerCase(); const cardCategory = card.dataset.category; const cardLanguage = card.dataset.language; const matchesSearch = title.includes(search); const matchesCategory = !category || cardCategory === category; const matchesLanguage = !language || cardLanguage === language; card.style.display = (matchesSearch && matchesCategory && matchesLanguage) ? "block" : "none"; }); }
